508  bool isCircleToCircleCollision(IND_Vector2 &pP1,
509  int pRadius1,
510  IND_Vector2 &pP2,
511  int pRadius2) {
512  // h^2 = x^2 + y^2 Pythagoras :D
513 
514  // x^2
515  double mDeltaXSquared = pP1._x - pP2._x;
516  mDeltaXSquared *= mDeltaXSquared;
517 
518  // y^2
519  double mDeltaYSquared = pP1._y - pP2._y;
520  mDeltaYSquared *= mDeltaYSquared;
521 
522  // Adding radius^2
523  double mSumRadiiSquared = pRadius1 + pRadius2;
524  mSumRadiiSquared *= mSumRadiiSquared;
525 
526  // If that result is greater than h^2, so less than x^2 + y^2
527  // then there is a collision
528  if (mDeltaXSquared + mDeltaYSquared <= mSumRadiiSquared)
529  return 1;
530  else
531  return 0;
532  }

570  static bool isPointInsideTriangle(IND_Vector2 &p,
571  IND_Vector2 &a,
572  IND_Vector2 &b,
573  IND_Vector2 &c) {
574  // Compute vectors
575  IND_Vector2 v0 = c - a;
576  IND_Vector2 v1 = b - a;
577  IND_Vector2 v2 = p - a;
578 
579  // Compute dot products
580  float dot00 = v0.dotProduct(v0);
581  float dot01 = v0.dotProduct(v1);
582  float dot02 = v0.dotProduct(v2);
583  float dot11 = v1.dotProduct(v1);
584  float dot12 = v1.dotProduct(v2);
585 
586  // Compute barycentric coordinates
587  float invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
588  float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
589  float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
590 
591  // Check if point is in triangle
592  return (u > 0) && (v > 0) && (u + v < 1);
593  }

604  double pointToLineDistance(IND_Vector2 &pA,
605  IND_Vector2 &pB,
606  IND_Vector2 &pC,
607  bool pIsSegment) {
608  IND_Vector2 ab (pB._x - pA._x, pB._y - pA._y);
609  IND_Vector2 ac (pC._x - pA._x, pC._y - pA._y);
610 
611  double mDist (ab.crossProduct(ac) / pA.distance(pB));
612 
613  if (pIsSegment) {
614  IND_Vector2 bc (pC._x - pB._x, pC._y - pB._y);
615  IND_Vector2 ba (pA._x - pB._x, pA._y - pB._y);
616 
617  float mDot1 (ab.dotProduct(bc));
618  if (mDot1 > 0.0f) {
619  return pB.distance(pC);
620  }
621  float mDot2 (ba.dotProduct(ac));
622  if (mDot2 > 0.0f) {
623  return pA.distance(pC);
624  }
625  }
626 
627  return fabs(mDist);
628  }

640  bool isSegmentIntersection(IND_Vector2 &a,
641  IND_Vector2 &b,
642  IND_Vector2 &c,
643  IND_Vector2 &d) {
644  float mGradAB, mGradCD, mYcptAB, mYcptCD, mInterceptX, mInterceptY;
645 
646  // In order to avoid divisions by zero
647  if (a._y == b._y)
648  b._y += 0.0001f;
649 
650  if (a._x == b._x)
651  b._x += 0.0001f;
652 
653  if (c._y == d._y)
654  d._y += 0.0001f;
655 
656  if (c._x == d._x)
657  d._x += 0.0001f;
658 
659  // Calculates the intersection between the two lines
660  mGradAB = (a._y - b._y) / (a._x - b._x);
661  mGradCD = (c._y - d._y) / (c._x - d._x);
662 
663  mYcptAB = a._y - a._x * mGradAB;
664  mYcptCD = c._y - c._x * mGradCD;
665  mInterceptX = (mYcptAB - mYcptCD) / (mGradCD - mGradAB);
666  mInterceptY = (mYcptAB - (mGradAB * mYcptCD) / mGradCD) / (1 - mGradAB / mGradCD);
667 
668  // Checking in the intersection is inside the segment
669  if (!((mInterceptX >= a._x && mInterceptX <= b._x) || (mInterceptX >= b._x && mInterceptX <= a._x)))
670  return 0;
671 
672  if (!((mInterceptY >= a._y && mInterceptY <= b._y) || (mInterceptY >= b._y && mInterceptY <= a._y)))
673  return 0;
674 
675  if (!((mInterceptX >= c._x && mInterceptX <= d._x) || (mInterceptX >= d._x && mInterceptX <= c._x)))
676  return 0;
677 
678  if (!((mInterceptY >= c._y && mInterceptY <= d._y) || (mInterceptY >= d._y && mInterceptY <= c._y)))
679  return 0;
680 
681  return 1;
682  }
